第一周-防抖和节流函数

JavaScript 前端
发布日期:   2020-07-10

问题引入

<style>
    .container {
        width: 100%;
        height: 200px;
        background-color: black;
        text-align: center;
        line-height: 200px;
        color: white;
    }
</style>
<body>
    <div class="container"></div>
</body>
<script>
    window.onload = function () {
        let count = 0;
        let container = document.querySelector(".container");
        container.onmousemove = function () {
            container.innerHTML = count++;
        };
    };
</script>
  • 鼠标滑过div时，内部的数值会不断增大（大约300）
  • 说明事件处理函数执行了300多次
    • 如果时较为复杂的函数或者时Ajax请求，会让浏览器运行速度急剧下滑甚至奔溃或者服务器崩溃

存在以下两种解决方案：防抖和节流

工具库

  • underscore

    一个JavaScript实用库，提供了一整套函数式编程的实用功能，但是没有扩展任何JavaScript内置对象

    • 内置了防抖函数（debounce）和节流函数（throttle）
    • 中文文档 https://www.html.cn/doc/underscore/

防抖函数

原理

  • 事件响应函数在一段时间后才执行，如果在这段时间内再次调用，则重新计算等待事件

    • 执行方式一

      时间结束后，如果在这段时间内没有调用则执行

    • 执行方式二

      立即执行，直到时间结束这段时间内没有调用，则下一次继续立即执行

代码

function debounce(func, wait, immediate) {
    //func为事件触发函数，wait是等待时间，immediate是否立即执行
    let timer//声明事件队列标识符
    let debounced = function () {
        let args = arguments;//保存参数
        let context = this;//保存上下文环境
        clearTimeout(timer)//清除事件队列
        if (immediate) {
            let callNow = !timer;//标记是否立即执行
            /*
            1.timer是undefined，则事件第一次执行或者等待时间已经过去
            2.timer不是undefined，则等待时间没过，需要重新计时
            */
            timer = setTimeout(() => {
                timer = undefined
            }, wait)//添加事件队列
            if (callNow) {
                func.apply(context, args)
            }
        } else {
            timer = setTimeout(function () {
                func.apply(context, args)
            }, wait)
            //添加事件队列
        }
    };
    debounced.cancel = function () {
        clearTimeout(timer)
        timer = null
    }
    //提供防抖取消操作
    return debounced;//返回防抖函数
}
};

解析：

  • 利用JavaScript事件队列的特性
    • 永远保证事件队列中只有一个异步操作或者标识符（针对一个事件）
    • 则每一次触发事件都先清除定时器（去除事件队列中的异步操作）

适用

  • scroll事件
  • 搜索框输入查询
  • 表单验证
  • 按钮提交
  • 浏览器的resize事件

节流函数

原理

  • 如果你持续触发事件，每隔一段时间，只执行一次事件
    • 事件触发立即执行，同时有结尾事件
    • 事件触发立即执行，无结尾事件
    • 事件不立即执行，有结尾事件
    • 事件不立即执行，无结尾事件，但是无法达成 （TODO：研究下）

代码

function throttle(func, wait, options) {
    let timer,
        old = 0;
    return function () {
        let context = this;//保存上下文
        let args = arguments;//保存参数
        //时间戳写法
        //效果是事件立即执行，没有结尾事件
        let now = Date.now();//保存当前时间
        if (options.leading === false && !old) {
            old = now;
        }//决定事件是否立即执行
        if (now - old > wait) {
            if (timer) {
                clearTimeout(timer);
                timer = undefined;
            }//防止事件重复执行,影响定时器写法
            func.apply(context, args);
            old = now;//刷新时间
        }
        //定时器写法
        //效果是事件不立即执行，有结尾事件
        if (!timer && options.trailing !== false) {
            timer = setTimeout(() => {
                old = Date.now();//防止事件重复执行，影响时间戳写法
                func.apply(this, args);
                timer = undefined;
            }, wait);
        }
    };
}
  • options中两个选项
    • leading 布尔值，是否立即执行事件
    • trailing 布尔值，是否添加结尾事件
  • 定时器写法和时间戳写法都可以做到函数节流，只是开始和结尾效果不同，这里是结合两者的改良型

适用

  • DOM元素的拖拽功能实现
  • 射击游戏类
  • 计算鼠标移动距离
  • 监听scroll滚动

防抖与节流的区别

例如：监听scroll滚动事件

  • 防抖函数会让滚动事件在用户结束滚动后执行
  • 节流函数会让用户在滚动的过程中隔一段时间就执行
